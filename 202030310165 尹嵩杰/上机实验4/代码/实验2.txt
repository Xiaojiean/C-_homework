#include<iostream>#include<string>using namespace std;class Student {public:	static float m_ClassMoney;	string sname;	void InitStudent(const char name[])	{		sname = name;	}	void ShowMoney()	{		cout <<sname<<"使用班费后 "<< "班费还剩余:" << m_ClassMoney << endl;	}	void ExpendMoney(float money)	{		m_ClassMoney = m_ClassMoney - money;	}};float Student::m_ClassMoney = 1000;int main(){	Student A, B, C;	A.InitStudent("A");	B.InitStudent("B");	C.InitStudent("C");	A.ExpendMoney(50);	A.ShowMoney();	B.ExpendMoney(98.5);	B.ShowMoney();	C.ExpendMoney(500.53);	C.ShowMoney();	return 0;}
